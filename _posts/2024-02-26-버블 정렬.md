  <script>
    document.addEventListener("DOMContentLoaded", function() {
      const arrayContainer = document.getElementById('array');
      const startBtn = document.getElementById('startBtn');
      
      let array = [4, 2, 7, 1, 3, 9, 5]; // 초기 배열

      // 배열 시각화 업데이트
      function updateArray() {
        arrayContainer.innerHTML = array.map(item => `<div class="array-item">${item}</div>`).join('');
      }

      // 버블 정렬 알고리즘
      async function bubbleSort() {
        let len = array.length;
        for (let i = 0; i < len; i++) {
          for (let j = 0; j < len - 1 - i; j++) {
            // 비교
            arrayContainer.children[j].style.backgroundColor = '#ff0000';
            arrayContainer.children[j + 1].style.backgroundColor = '#ff0000';
            await sleep(500); // 애니메이션 딜레이

            if (array[j] > array[j + 1]) {
              // 교환
              let temp = array[j];
              array[j] = array[j + 1];
              array[j + 1] = temp;
              updateArray(); // 배열 업데이트
            }

            // 원래 색상으로 되돌리기
            arrayContainer.children[j].style.backgroundColor = '#f0f0f0';
            arrayContainer.children[j + 1].style.backgroundColor = '#f0f0f0';
          }
        }
      }

      // 비동기 딜레이 함수
      function sleep(ms) {
        return new Promise(resolve => setTimeout(resolve, ms));
      }

      // Start 버튼 클릭 시
      startBtn.addEventListener("click", function() {
        bubbleSort();
      });
    });
  </script>
